//선택자 const searchBox = document.querySelector(".search__box input"); //input 박스 const searchList = document.querySelectorAll(".search__list ul li"); //목록 리스트 const searchInfo = document.querySelector(".search__info .num"); //전체 갯수 let count = 0; //전체 갯수 구하기 searchInfo.textContent = searchList.length; //검색 영역 searchBox.addEventListener("keyup", () => { const searchWord = searchBox.value; //사용자가 입력한 키워드 searchList.forEach(el => { const cssName = el.dataset.name; if (cssName.indexOf(searchWord)) { el.classList.add("hide"); } else { el.classList.remove("hide"); //indexOf()값이 0(false)이되는 값만 hide를 지워줍니다. count++ } searchInfo.textContent = count; }); count = 0; });
